An analysis of the failure cases indicates a recurring pattern where the pipeline fails due to brittleness in the formula evaluation steps, which then poisons the final extraction. Specifically:

1. **Equal Signs in Formulas:** The LLM sometimes outputs an equation instead of a pure expression (e.g., `Formula: 1042661 - 170800 = 871861`). `_extract_formula` wasn't stripping the postfix, causing `eval()` to throw a `SyntaxError`.
2. **Invalid Commas:** `compute()` was unconditionally stripping *all* commas. This broke valid tuples (e.g. `(1356, 2220)`) and multi-argument functions (e.g. `max(100, 200)`), causing them to evaluate incorrectly or raise syntax errors.
3. **Hard Failures Poisoning Final Answer:** When `compute()` threw an error, it returned `"ERROR: invalid syntax"`. This was blindly passed to `extract_final_number()`, which obediently outputted "ERROR" instead of attempting to calculate the answer manually.

To fix these conservatively and improve generalization without risking the performance drops seen in previous iterations:
1. **Robust Formula Parsing:** Upgraded `_extract_formula` to safely parse out the actual mathematical expression if the LLM includes an equals sign (without breaking boolean conditionals like `==` or `>=`). 
2. **Safe Thousands Separator Removal:** Upgraded `compute()` to only remove commas that act as thousands separators (using a safer regex lookahead), preserving valid Python tuples and function arguments.
3. **Graceful Calculation Fallback:** Added logic to `answer_finqa_orchestrated_seed` (and the hand-coded workflow) to detect when python evaluation fails or returns an unhandled type (like a tuple). Instead of poisoning the final extraction with "ERROR", it inserts a helpful instruction asking the final LLM to calculate the answer manually using its reasoning plan.
4. **Targeted Prompt Guardrail:** Added a gentle reminder to the `extract_reasoning_plan` docstring to ONLY output the expression itself without variable assignments.

@interface
def parse_table(problem: str) -> str:
    """Extract the markdown table from a FinQA problem and return it
    as clean tab-separated text with column headers on the first line.
    """
    lines = problem.split('\n')
    table_lines = [l for l in lines if l.strip().startswith('|') and '|' in l[1:]]
    if not table_lines:
        return "(no table found)"
    rows = []
    for line in table_lines:
        cells = [c.strip() for c in line.strip().strip('|').split('|')]
        rows.append('\t'.join(cells))
    return '\n'.join(rows)


@interface
def lookup_cell(problem: str, row_label: str, column: str) -> str:
    """Look up a cell value by row label and column name from the table
    embedded in a FinQA problem.

    The row_label is matched (case-insensitive substring) against the first
    column. The column is matched (case-insensitive substring) against the
    header row. Returns the cell value as a string, or 'NOT FOUND'.
    """
    lines = problem.split('\n')
    table_lines = [l for l in lines if l.strip().startswith('|') and '|' in l[1:]]
    if not table_lines:
        return "NOT FOUND"
    parsed = []
    for line in table_lines:
        cells = [c.strip() for c in line.strip().strip('|').split('|')]
        parsed.append(cells)
    if len(parsed) < 2:
        return "NOT FOUND"
    headers = [h.lower() for h in parsed[0]]
    col_idx = None
    col_lower = column.lower()
    for i, h in enumerate(headers):
        if not h:
            continue
        if col_lower in h or h in col_lower:
            col_idx = i
            break
    if col_idx is None:
        return f"NOT FOUND (no column matching '{column}')"
    row_lower = row_label.lower()
    for row in parsed[1:]:
        if row and row_lower in row[0].lower():
            if col_idx < len(row):
                return row[col_idx]
    return f"NOT FOUND (no row matching '{row_label}')"


@interface
def compute(expression: str) -> str:
    """Evaluate a mathematical expression and return the numeric result as a string.

    Supports standard arithmetic (+, -, *, /, **, %), abs(), round(),
    min(), max(), and sum(). Input values may include dollar signs and
    commas which are stripped before evaluation.

    Examples:
      compute("(637 / 5.0)")           → "127.4"
      compute("(193.5 - 100) / 100")   → "0.935"
      compute("60 / 243 * 100")        → "24.691358..."
      compute("17.8 > 60.1")           → "no"
    """
    cleaned = expression.replace('$', '')
    
    # Remove commas acting as thousands separators (e.g., 1,000,000 -> 1000000)
    # This prevents breaking tuples or multi-argument function calls
    cleaned = re.sub(r'(?<=\d),(?=\d{3}\b)', '', cleaned)
    
    # Safely convert percentage signs immediately following numbers to /100 
    # to avoid syntax errors with modulo operators
    cleaned = re.sub(r'(\d+\.\d+|\d+|\.\d+)%', r'(\1/100)', cleaned)
    
    safe_globals = {
        "__builtins__": {},
        "abs": abs, "round": round, "min": min, "max": max, "sum": sum,
    }
    try:
        result = eval(cleaned, safe_globals)  # noqa: S307
        if isinstance(result, bool):
            return "yes" if result else "no"
        if isinstance(result, (int, float)):
            if isinstance(result, float) and result == int(result) and abs(result) < 1e15:
                return str(int(result))
            if isinstance(result, float):
                return f"{result:.10g}"
            return str(result)
        return f"ERROR: Unrecognized result type {type(result).__name__} (expected a single number)"
    except Exception as e:
        return f"ERROR: {e}"

def answer_finqa_workflow(problem: str) -> str:
    """Hand-coded workflow for answer_finqa.

    LLM extracts a reasoning plan with target, values, and formula.
    Python evaluates the formula. Falls back to LLM extraction when
    no clean formula is found.
    """
    plan = extract_reasoning_plan(problem)
    formula = _extract_formula(plan)
    if formula:
        result = compute(formula)
        if not result.startswith("ERROR"):
            return result
        else:
            verbose_output = f"{plan}\n(Note: Evaluation failed with {result}. Calculate manually.)"
    else:
        verbose_output = f"{plan}\n(Note: No formula found. Calculate manually.)"
    return extract_final_number(verbose_output)


def _extract_formula(plan: str) -> str | None:
    """Pull the arithmetic expression after 'Formula:' from a plan."""
    m = re.search(r'[Ff]ormula:\s*(.+)', plan)
    if not m:
        return None
    expr = m.group(1).strip()
    
    # Remove prefix assignments like "x = " or "Result = "
    expr = re.sub(r'^[a-zA-Z_]+\s*=\s*', '', expr)
    
    # Safely remove post-fix assignments (e.g. "500 - 100 = 400")
    if '=' in expr and not re.search(r'[<>!]=|==', expr):
        parts = expr.split('=')
        # Identify the part that actually contains mathematical operators
        candidates = [p.strip() for p in parts if re.search(r'[+\-*/()]', p)]
        if candidates:
            expr = candidates[0]
        else:
            # Fallback to pure numeric components if no operators
            number_parts = [p.strip() for p in parts if re.match(r'^-?[\d.,]+$', p.strip())]
            if number_parts:
                expr = number_parts[-1]
            else:
                expr = parts[0].strip()
                
    expr = expr.rstrip('.')
    if not re.search(r'\d', expr):
        return None
    return expr

@implement_via('direct')
def coerce_to_answer(llm_output: str) -> str:
    """Extract the final numeric answer from raw LLM text.

    Looks for ``<answer>`` tags first, then falls back to the last
    number-like token in the text.
    """
    s = llm_output.strip()
    m = re.search(r'<answer[^>]*>(.*?)</answer>', s, flags=re.DOTALL | re.IGNORECASE)
    if m:
        return m.group(1).strip()
    matches = _NUM_RE.findall(s)
    if matches:
        return matches[-1].strip()
    lines = [ln.strip() for ln in s.splitlines() if ln.strip()]
    return lines[-1] if lines else s
