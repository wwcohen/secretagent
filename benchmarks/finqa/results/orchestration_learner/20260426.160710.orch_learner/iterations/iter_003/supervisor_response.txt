An analysis of the failure cases indicates two primary issues causing regressions in the previous iteration:

1. **Overfitted Prompt Instructions:** The previously added instruction `"DO NOT multiply by 100"` was overly restrictive. While it helped standard decimal cases, it broke many FinQA examples where the ground truth *expected* a percentage in absolute terms (e.g., `24.69` instead of `0.2469`). Removing this strict rule and letting the model adapt format to the question leads to much better generalization. 
2. **Formula Extraction & Evaluation Brittleness:** 
    - The LLM sometimes outputs formulas as assignments (e.g., `Formula: 5 + 3 = 8` or `Formula: x = 5 + 3`), causing syntax errors in `eval`.
    - The LLM sometimes uses commas as thousands separators (e.g. `1,000 / 2,000`), which causes `eval` to treat them as invalid tuples instead of math.
    - When `eval` fails, it currently poisons the extraction context by injecting `Calculated: ... = ERROR: invalid syntax`, tricking the final extractor into outputting the word `"ERROR"`.

To address this conservatively without breaking working cases:
1. **Loosen Formatting Rules:** Revert the overly rigid rule about percentages in both prompt docstrings and replace it with a rule to format numbers exactly as the question implies. 
2. **Robust Comma Handling:** Update `compute()` to use a targeted regex that *only* removes commas acting as thousands separators (leaving function calls like `max(1, 2)` perfectly intact).
3. **Robust Equation Parsing:** Update `_extract_formula()` to safely extract the actual math expression from assignments (e.g., pulling `5 + 3` out of `5 + 3 = 8`).
4. **Graceful Fallback:** If `eval` throws an error or no formula is found, intercept the error and instead prompt the final `extract_final_number` LLM to manually calculate the answer based on the reasoning plan.

@interface
def parse_table(problem: str) -> str:
    """Extract the markdown table from a FinQA problem and return it
    as clean tab-separated text with column headers on the first line.
    """
    lines = problem.split('\n')
    table_lines = [l for l in lines if l.strip().startswith('|') and '|' in l[1:]]
    if not table_lines:
        return "(no table found)"
    rows = []
    for line in table_lines:
        cells = [c.strip() for c in line.strip().strip('|').split('|')]
        rows.append('\t'.join(cells))
    return '\n'.join(rows)


@interface
def lookup_cell(problem: str, row_label: str, column: str) -> str:
    """Look up a cell value by row label and column name from the table
    embedded in a FinQA problem.

    The row_label is matched (case-insensitive substring) against the first
    column. The column is matched (case-insensitive substring) against the
    header row. Returns the cell value as a string, or 'NOT FOUND'.
    """
    lines = problem.split('\n')
    table_lines = [l for l in lines if l.strip().startswith('|') and '|' in l[1:]]
    if not table_lines:
        return "NOT FOUND"
    parsed = []
    for line in table_lines:
        cells = [c.strip() for c in line.strip().strip('|').split('|')]
        parsed.append(cells)
    if len(parsed) < 2:
        return "NOT FOUND"
    headers = [h.lower() for h in parsed[0]]
    col_idx = None
    col_lower = column.lower()
    for i, h in enumerate(headers):
        if not h:
            continue
        if col_lower in h or h in col_lower:
            col_idx = i
            break
    if col_idx is None:
        return f"NOT FOUND (no column matching '{column}')"
    row_lower = row_label.lower()
    for row in parsed[1:]:
        if row and row_lower in row[0].lower():
            if col_idx < len(row):
                return row[col_idx]
    return f"NOT FOUND (no row matching '{row_label}')"


@interface
def compute(expression: str) -> str:
    """Evaluate a mathematical expression and return the numeric result as a string.

    Supports standard arithmetic (+, -, *, /, **, %), abs(), round(),
    min(), max(), and sum(). Input values may include dollar signs and
    commas which are stripped before evaluation.

    Examples:
      compute("(637 / 5.0)")           → "127.4"
      compute("(193.5 - 100) / 100")   → "0.935"
      compute("60 / 243 * 100")        → "24.691358..."
      compute("17.8 > 60.1")           → "no"
    """
    cleaned = expression.replace('$', '')
    # Safely remove commas used as thousands separators (e.g., 1,000 -> 1000)
    cleaned = re.sub(r'(?<=\d),(?=\d{3}\b)', '', cleaned)
    cleaned = re.sub(r'(?<=\d),(?=\d{3}\b)', '', cleaned)
    
    # Safely convert percentage signs immediately following numbers to /100.0 
    cleaned = re.sub(r'(\d+\.\d+|\d+|\.\d+)\s*%', r'(\1/100.0)', cleaned)
    
    safe_globals = {
        "__builtins__": {},
        "abs": abs, "round": round, "min": min, "max": max, "sum": sum,
    }
    try:
        result = eval(cleaned, safe_globals)  # noqa: S307
        if isinstance(result, bool):
            return "yes" if result else "no"
        if isinstance(result, float):
            if result == int(result) and abs(result) < 1e15:
                return str(int(result))
            return f"{result:.10g}"
        return str(result)
    except Exception as e:
        return f"ERROR: {e}"

def answer_finqa_workflow(problem: str) -> str:
    """Hand-coded workflow for answer_finqa.

    LLM extracts a reasoning plan with target, values, and formula.
    Python evaluates the formula. Falls back to LLM extraction when
    no clean formula is found.
    """
    plan = extract_reasoning_plan(problem)
    formula = _extract_formula(plan)
    if formula:
        result = compute(formula)
        if not result.startswith("ERROR"):
            return result
    return extract_final_number(plan)


def _extract_formula(plan: str) -> str | None:
    """Pull the arithmetic expression after 'Formula:' from a plan."""
    m = re.search(r'[Ff]ormula:\s*(.+)', plan)
    if not m:
        return None
    expr = m.group(1).strip()
    
    # Handle equations
    if '=' in expr and not any(op in expr for op in ['==', '>=', '<=', '!=', '>', '<']):
        parts = expr.split('=', 1)
        if re.search(r'[\+\-\*/]', parts[0]):
            expr = parts[0].strip()
        elif re.search(r'[\+\-\*/]', parts[1]):
            expr = parts[1].strip()
        else:
            expr = parts[1].strip() if not re.search(r'\d', parts[0]) else parts[0].strip()
            
    expr = re.sub(r'^[a-zA-Z_]+\s*=\s*', '', expr)
    expr = expr.rstrip('.')
    
    if not re.search(r'\d', expr):
        return None
    return expr

_NUM_RE = re.compile(r'[$€£]?\s*-?\s*[\d,]+\.?\d*\s*%?')


@implement_via('direct')
def coerce_to_answer(llm_output: str) -> str:
    """Extract the final numeric answer from raw LLM text.

    Looks for ``<answer>`` tags first, then falls back to the last
    number-like token in the text.
    """
    s = llm_output.strip()
    m = re.search(r'<answer[^>]*>(.*?)</answer>', s, flags=re.DOTALL | re.IGNORECASE)
    if m:
        return m.group(1).strip()
    matches = _NUM_RE.findall(s)
    if matches:
        return matches[-1].strip()
    lines = [ln.strip() for ln in s.splitlines() if ln.strip()]
    return lines[-1] if lines else s
