def time_to_mins(t_str: str) -> int:
    t_str = t_str.strip()
    am_pm = t_str[-2:]
    time_part = t_str[:-2]
    h_str, m_str = time_part.split(':')

    if h == 12:

def mins_to_time(m_total: int) -> str:
    h = (m_total // 60) % 24
    m = m_total % 60
    am_pm = 'AM' if h < 12 else 'PM'
    h_disp = h % 12
    if h_disp == 0:
        h_disp = 12
    return f"{h_disp}:{m:02d}{am_pm}"
