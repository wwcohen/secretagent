def trip_planning_orchestrated_seed(prompt: str) -> str:
    try:
        # Parse N and total_days
        m = re.search(r'visit (\d+) European cities for (\d+) days', prompt)
        if not m:
            raise ValueError("Could not parse N and total_days")
        N_cities = int(m.group(1))
        total_days = int(m.group(2))
        
        # Parse durations
        durations = {}
        sentences = prompt.replace('\n', '. ').split('.')
        for s in sentences:
            s = s.strip()
            if not s: continue
            
            m1 = re.search(r'(?:visit|stay in) ([A-Z][a-zA-Z]+) for (\d+) days', s)
            if m1:
                durations[m1.group(1)] = int(m1.group(2))
                continue
            
            m2 = re.search(r'spend (\d+) days in ([A-Z][a-zA-Z]+)', s)
            if m2:
                durations[m2.group(2)] = int(m2.group(1))
                
        if len(durations) != N_cities:
            raise ValueError(f"Extracted {len(durations)} cities, expected {N_cities}")
            
        # Parse time windows
        time_windows = {}
        for s in sentences:
            s = s.strip()
            if not s: continue
            m3 = re.search(r'(?:between|From|During) day (\d+) (?:and|to) day (\d+)', s)
            if m3:
                start_day = int(m3.group(1))
                end_day = int(m3.group(2))
                for city in durations:
                    if re.search(rf'\b{city}\b', s):
                        time_windows[city] = (start_day, end_day)
                        break
                        
        # Parse flights
        adj = {city: set() for city in durations}
        flights_str = re.search(r'Here are the cities that have direct flights:\s*\n(.*?)(?:\n\n|\Z)', prompt, re.DOTALL)
        if not flights_str:
            raise ValueError("Could not parse flights block")
            
        flights_text = flights_str.group(1).strip()
        if flights_text.endswith('.'):
            flights_text = flights_text[:-1]
            
        flight_pairs = flights_text.split(', ')
        for pair in flight_pairs:
            # We must parse flights strictly as DIRECTED edges (Source -> Destination).
            # The phrasing is consistently "X and Y" or "from X to Y", meaning X -> Y.
            found = []
            for c in durations:
                match = re.search(rf'\b{c}\b', pair)
                if match:
                    found.append((match.start(), c))
            if len(found) == 2:
                found.sort()  # Sort by string match index to get Source then Destination
                u = found[0][1]
                v = found[1][1]
                adj[u].add(v)
                
        # DFS to find path
        all_cities = list(durations.keys())
        
        def dfs(path, current_day):
            if len(path) == N_cities:
                return path
                
            for nxt in all_cities:
                if nxt in path:
                    continue
                    
                if len(path) > 0 and nxt not in adj[path[-1]]:
                    continue
                    
                start_day = current_day
                end_day = current_day + durations[nxt] - 1
                
                if nxt in time_windows:
                    req_start, req_end = time_windows[nxt]
                    if start_day != req_start or end_day != req_end:
                        continue
                        
                res = dfs(path + [nxt], end_day)
                if res:
                    return res
                    
            return None

        path = None
        for start_city in all_cities:
            start_day = 1
            end_day = start_day + durations[start_city] - 1
            
            if start_city in time_windows:
                req_start, req_end = time_windows[start_city]
                if start_day != req_start or end_day != req_end:
                    continue
                    
            res = dfs([start_city], end_day)
            if res:
                path = res
                break
                
        if path is None:
            raise ValueError("No valid path found")
            
        # Format the result strictly
        lines = []
        lines.append(f"Here is the trip plan for visiting the {N_cities} European cities for {total_days} days:\n")
        
        current_day = 1
        for i, city in enumerate(path):
            dur = durations[city]
            start_day = current_day
            end_day = current_day + dur - 1
            
            if i == 0:
                lines.append(f"**Day {start_day}-{end_day}:** Arriving in {city} and visit {city} for {dur} days.")
            else:
                prev_city = path[i-1]
                lines.append(f"**Day {start_day}:** Fly from {prev_city} to {city}.")
                lines.append(f"**Day {start_day}-{end_day}:** Visit {city} for {dur} days.")
                
            current_day = end_day
            
        return "\n".join(lines)
        
    except Exception:
