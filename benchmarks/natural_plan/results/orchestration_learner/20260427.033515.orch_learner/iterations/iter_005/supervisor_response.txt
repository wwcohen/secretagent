An analysis of the failing cases reveals two critical issues that cascade into failures:

1. **LLM Omitting the Earliest Valid Slot:** In cases like `calendar_scheduling_example_334` and `150`, the LLM's `find_available_slots` correctly identifies some later valid slots but *entirely misses* the earliest valid slot. def _time_to_mins(t_str: str) -> int:
    try:
        parts = t_str.split(":")
        return int(parts[0]) * 60 + int(parts[1])
    except:
        return 0

def _mins_to_time(mins: int) -> str:
    h = mins // 60
    m = mins % 60
    return f"{h}:{m:02d}"

def _generate_slots_python(schedules_data: dict) -> list:
    duration = schedules_data.get("duration_minutes", 30)
    wh = schedules_data.get("working_hours", ["9:00", "17:00"])
    start_wh = _time_to_mins(wh[0])
    end_wh = _time_to_mins(wh[1])
    days = schedules_data.get("days", ["Monday"])
    
    valid_slots = []
    for day in days:
        for t in range(start_wh, end_wh - duration + 1, 30):
            slot_start = t
            slot_end = t + duration
            
            conflict = False
            for person, days_dict in schedules_data.get("schedules", {}).items():
                busy_intervals = days_dict.get(day, [])
                for b_start, b_end in busy_intervals:
                    b_start_m = _time_to_mins(b_start)
                    b_end_m = _time_to_mins(b_end)
                    if slot_start < b_end_m and slot_end > b_start_m:
                        conflict = True
                        break
                if conflict:
                    break
            
            if not conflict:
                valid_slots.append({
                    "day": day,
                    "start": _mins_to_time(slot_start),
                    "end": _mins_to_time(slot_end)
                })
    return valid_slots

def _filter_and_fallback_slots(schedules_json: str, slots_json: str) -> str:
    try:
        schedules_data = json.loads(schedules_json)
        
        # Always rely on the mathematically perfect list of valid slots.
        # This prevents the bug where the LLM misses the earliest valid slot.
        python_slots = _generate_slots_python(schedules_data)
        
        if python_slots:
            return json.dumps(python_slots)
            
    except Exception:
        pass
        
    # If parsing failed or Python found no slots (indicating hallucinated 
    # over-constraints in parse_schedules), fallback to the LLM's intuition.
    return slots_json
