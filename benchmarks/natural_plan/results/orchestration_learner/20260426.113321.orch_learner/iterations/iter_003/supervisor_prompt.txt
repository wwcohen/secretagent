"""Task-specific interfaces for NaturalPlan meeting planning.

Decomposition derived from LLM reasoning traces:
1. parse_meeting_info — extract locations, friends, travel times, availability
2. plan_visit_order — determine optimal visit order to maximize meetings
3. build_meeting_plan — simulate schedule step-by-step and format answer

from secretagent.core import interface
import json
import re

def parse_time(t_str):
    t_str = t_str.replace(" ", "").upper()
    period = t_str[-2:]
    time_part = t_str[:-2]
    h, m = map(int, time_part.split(':'))
    if h == 12:
        h = 0
    if period == 'PM':
        h += 12
    return h * 60 + m

def format_time(mins):
    h = mins // 60
    m = mins % 60
    period = "AM"
    if h >= 12:
        period = "PM"
        if h > 12:
            h -= 12
    if h == 0:
        h = 12
    return f"{h}:{m:02d}{period}"

def _parse_logic(prompt: str):
    dist_section = prompt.split("CONSTRAINTS:")[0]
    if "CONSTRAINTS:" in prompt:
        constraints_section = prompt.split("CONSTRAINTS:")[1]
    else:
        constraints_section = prompt
        
    distances = {}
    for match in re.finditer(r"([A-Za-z0-9\s'-]+) to ([A-Za-z0-9\s'-]+):\s*(\d+)\.", dist_section):
        src = match.group(1).strip()
        dst = match.group(2).strip()
        dist = int(match.group(3))
        distances.setdefault(src, {})[dst] = dist
        
    arr_match = re.search(r"You arrive at ([A-Za-z0-9\s'-]+) at (\d{1,2}:\d{2}\s*[AP]M)\.", constraints_section)
    if arr_match:
        start_loc = arr_match.group(1).strip()
        start_time_str = arr_match.group(2).strip()
    else:
        start_loc = ""
        start_time_str = "9:00AM"
        
    friends = []
    friend_pattern = r"([A-Za-z]+) will be at ([A-Za-z0-9\s'-]+) from (\d{1,2}:\d{2}\s*[AP]M) to (\d{1,2}:\d{2}\s*[AP]M)\.\s*You'd like to meet \1 for a minimum of (\d+) minutes\."
    for match in re.finditer(friend_pattern, constraints_section):
        friends.append({
            "name": match.group(1).strip(),
            "location": match.group(2).strip(),
            "available_from": match.group(3).strip(),
            "available_to": match.group(4).strip(),
            "duration_minutes": int(match.group(5))
        })
        
    return {
        "my_location": start_loc,
        "my_start_time": start_time_str,
        "friends": friends,
        "travel_times": {f"{k}->{k2}": v for k, d in distances.items() for k2, v in d.items()}
    }

def _solve_path(parsed):
    start_loc = parsed["my_location"]
    start_time = parse_time(parsed["my_start_time"])
    
    distances = {}
    for k, v in parsed["travel_times"].items():
        src, dst = k.split("->")
        distances.setdefault(src, {})[dst] = v
        
    friends = parsed["friends"]
    for f in friends:
        f['start'] = parse_time(f['available_from'])
        f['end'] = parse_time(f['available_to'])
        f['duration'] = f['duration_minutes']
        
    n = len(friends)
    dp = {}
    
    for i in range(n):
        loc = friends[i]['location']
        t_time = distances.get(start_loc, {}).get(loc, 0) if start_loc != loc else 0
        arr_time = start_time + t_time
        m_start = max(arr_time, friends[i]['start'])
        m_end = m_start + friends[i]['duration']
        if m_end <= friends[i]['end']:
            dp[(1 << i, i)] = (m_end, -1, m_start, arr_time, t_time, t_time)
            
    for mask in range(1, 1 << n):
        for i in range(n):
            if (mask & (1 << i)) == 0:
                continue
            if (mask, i) not in dp:
                continue
                
            curr_end, _, _, _, _, curr_tot = dp[(mask, i)]
            curr_loc = friends[i]['location']
            
            for j in range(n):
                if (mask & (1 << j)) != 0:
                    continue
                    
                loc = friends[j]['location']
                t_time = distances.get(curr_loc, {}).get(loc, 0) if curr_loc != loc else 0
                arr_time = curr_end + t_time
                m_start = max(arr_time, friends[j]['start'])
                m_end = m_start + friends[j]['duration']
                tot_travel = curr_tot + t_time
                
                if m_end <= friends[j]['end']:
                    new_mask = mask | (1 << j)
                    update = False
                    if (new_mask, j) not in dp:
                        update = True
                    else:
                        old_m_end, _, _, _, _, old_tot = dp[(new_mask, j)]
                        if m_end < old_m_end:
                            update = True
                        elif m_end == old_m_end and tot_travel < old_tot:
                            update = True
                            
                    if update:
                        dp[(new_mask, j)] = (m_end, i, m_start, arr_time, t_time, tot_travel)
                        
    best_mask = 0
    best_last = -1
    best_bits = -1
    best_end = float('inf')
    best_tot = float('inf')
    
    for (mask, i), val in dp.items():
        bits = bin(mask).count('1')
        m_end = val[0]
        tot_travel = val[5]
        if bits > best_bits:
            best_bits = bits
            best_end = m_end
            best_tot = tot_travel
            best_mask = mask
            best_last = i
        elif bits == best_bits:
            if m_end < best_end:
                best_end = m_end
                best_tot = tot_travel
                best_mask = mask
                best_last = i
            elif m_end == best_end and tot_travel < best_tot:
                best_tot = tot_travel
                best_mask = mask
                best_last = i
                
    path = []
    if best_bits == -1:
        return path
        
    curr_mask = best_mask
    curr_i = best_last
    
    while curr_i != -1:
        val = dp[(curr_mask, curr_i)]
        m_end, prev_i, m_start, arr_time, t_time, _ = val
        path.append({
            'friend': friends[curr_i],
            'travel_time': t_time,
            'arrive_time': arr_time,
            'meet_start': m_start,
            'meet_end': m_end
        })
        curr_mask = curr_mask ^ (1 << curr_i)
        curr_i = prev_i
        
    path.reverse()
    return path

def solve_meeting(prompt: str) -> str:
    parsed = _parse_logic(prompt)
    path = _solve_path(parsed)
    
    start_loc = parsed["my_location"]
    start_time = parse_time(parsed["my_start_time"])
    
    ans = ["SOLUTION:"]
    ans.append(f"You start at {start_loc} at {format_time(start_time)}.")
    for step in path:
        f = step['friend']
        if step['travel_time'] > 0:
            ans.append(f"You travel to {f['location']} in {step['travel_time']} minutes and arrive at {format_time(step['arrive_time'])}.")
        if step['meet_start'] > step['arrive_time']:
            ans.append(f"You wait until {format_time(step['meet_start'])}.")
        ans.append(f"You meet {f['name']} for {f['duration']} minutes from {format_time(step['meet_start'])} to {format_time(step['meet_end'])}.")
        
    return " ".join(ans)

@interface
def parse_meeting_info(problem_text: str) -> str:
    """Parse a meeting-planning problem into structured JSON."""
    return json.dumps(_parse_logic(problem_text))

@interface
def plan_visit_order(problem_text: str, info_json: str) -> str:
    """Determine the optimal order of friends to visit to maximize meetings."""
    try:
        parsed = json.loads(info_json)
        path = _solve_path(parsed)
        return json.dumps([step['friend']['name'] for step in path])
    except:
        path = _solve_path(_parse_logic(problem_text))
        return json.dumps([step['friend']['name'] for step in path])

@interface
def build_meeting_plan(problem_text: str, info_json: str, order_json: str) -> str:
    """Build a step-by-step meeting schedule and format the final answer."""
    return solve_meeting(problem_text)

@interface
def meeting_planning(prompt: str) -> str:
    """Solve a meeting-planning problem."""
    return solve_meeting(prompt)

def meeting_workflow(prompt: str) -> str:
    return solve_meeting(prompt)
