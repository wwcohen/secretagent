"""Tools for the geometric_shapes benchmark.

The task: given an SVG path element, identify which geometric shape it draws,
returning the correct multiple-choice letter (e.g. "(J)").
"""

import math
from collections import defaultdict
from typing import List, Optional, Tuple

from secretagent.core import interface, implement_via

# ── path normalization ──────────────────────────────────────────────────────

def _round_pt(x: float, y: float, decimals: int = 2) -> Tuple[float, float]:
    return (round(x, decimals), round(y, decimals))

def _parse_coord(s: str) -> Tuple[float, float]:
    """Parse 'x,y' or 'x y' into a float tuple."""
    parts = s.replace(',', ' ').split()
    return (float(parts[0]), float(parts[1]))

def _segments_from_commands(commands: List[str]) -> List[Tuple[Tuple[float, float], Tuple[float, float]]]:
    """Extract line segments from a list of SVG commands.

    Returns a list of (start, end) point pairs, one per L command.
    """
    segments = []
    current = None
    for cmd in commands:
        cmd = cmd.strip()
        if cmd.upper().startswith('M'):
            current = _round_pt(*_parse_coord(cmd[1:].strip()))
        elif cmd.upper().startswith('L'):
            end = _round_pt(*_parse_coord(cmd[1:].strip()))
            if current is not None:
                segments.append((current, end))
            current = end
    return segments

def _find_eulerian_path(adj, degree):
    """Find an Eulerian path using Hierholzer's algorithm on a multigraph.

    adj: dict mapping node -> list of (neighbor, edge_index)
    degree: dict mapping node -> degree (number of edges)

    Returns a list of nodes forming the path, or None if no Eulerian path exists.
    """
    # find start node: prefer an odd-degree node, else any node with edges
    odd_nodes = [n for n, d in degree.items() if d % 2 == 1]
    if len(odd_nodes) > 2:
        return None  # no Eulerian path possible
    start = odd_nodes[0] if odd_nodes else next((n for n, d in degree.items() if d > 0), None)
    if start is None:
        return None

    stack = [start]
    path = []
    used = set()
    while stack:
        v = stack[-1]
        found = False
        while adj[v]:
            u, idx = adj[v].pop()
            if idx not in used:
                used.add(idx)
                # also remove from the other side's list lazily (handled by used set)
                stack.append(u)
                found = True
                break
        if not found:
            path.append(stack.pop())
    path.reverse()
    return path

def normalize_path(commands: List[str]) -> List[str]:
    """Rearrange SVG commands into continuous chains where possible.

    Treats each L command as an undirected edge between two points, then
    finds a continuous traversal (Eulerian path) of those edges. Falls back
    to separate M-started chains for disconnected components.
    """
    segments = _segments_from_commands(commands)
    if not segments:
        return commands

    # build adjacency for Eulerian path
    adj = defaultdict(list)
    degree = defaultdict(int)
    for idx, (a, b) in enumerate(segments):
        adj[a].append((b, idx))
        adj[b].append((a, idx))
        degree[a] += 1
        degree[b] += 1

    # try to find an Eulerian path over the whole graph
    path = _find_eulerian_path(adj, degree)
    if path is not None and len(path) == len(segments) + 1:
        result = [f'M {path[0][0]},{path[0][1]}']
        for pt in path[1:]:
            result.append(f'L {pt[0]},{pt[1]}')
        return result

    # fallback: chain segments greedily per connected component
    point_to_segs = defaultdict(list)
    for idx, (a, b) in enumerate(segments):
        point_to_segs[a].append(idx)
        point_to_segs[b].append(idx)

    used = [False] * len(segments)
    result = []

    for start_idx in range(len(segments)):
        if used[start_idx]:
            continue
        # start a new chain from this segment
        a, b = segments[start_idx]
        used[start_idx] = True
        chain = [a, b]
        # extend forward
        while True:
            tip = chain[-1]
            found = False
            for idx in point_to_segs[tip]:
                if not used[idx]:
                    used[idx] = True
                    sa, sb = segments[idx]
                    chain.append(sb if sa == tip else sa)
                    found = True
                    break
            if not found:
                break
        result.append(f'M {chain[0][0]},{chain[0][1]}')
        for pt in chain[1:]:
            result.append(f'L {pt[0]},{pt[1]}')

    return result

def _predict_shape_from_path(input: str) -> Optional[str]:
    """Pure Python mathematical analysis of the SVG path. Eliminates LLM
    hallucination on vertex counting and angle identification by determining the
    exact geometric class of the path.
    """
    try:
        path, options_list = extract_path_and_options(input)
        commands = decompose_path(path)
    except Exception:
        return None
        
    pts = []
    for cmd in commands:
        cmd_strip = cmd.strip().upper()
        if cmd_strip.startswith('M') or cmd_strip.startswith('L'):
            try:
                pt = _parse_coord(cmd_strip[1:].strip())
                # prevent adjacent duplicates
                if not pts or math.hypot(pts[-1][0]-pt[0], pts[-1][1]-pt[1]) > 0.1:
                    pts.append(pt)
            except Exception:
                pass
                
    # remove the closing point if it simply matches the start
    if len(pts) > 1 and math.hypot(pts[0][0]-pts[-1][0], pts[0][1]-pts[-1][1]) < 0.1:
        pts.pop()
        
    n_points = len(pts)
    
    has_arcs = any(cmd.strip().upper().startswith('A') for cmd in commands)
    num_arcs = sum(1 for cmd in commands if cmd.strip().upper().startswith('A'))
    
    predicted_shape = None
    
    if has_arcs:
        if num_arcs == 1 and n_points >= 2:
            predicted_shape = "sector"
        elif num_arcs >= 2:
            arc_cmds = [cmd for cmd in commands if cmd.strip().upper().startswith('A')]
            is_ellipse = False
            for acmd in arc_cmds:
                parts = acmd[1:].strip().replace(',', ' ').split()
                if len(parts) >= 3:
                    try:
                        rx, ry, rot = float(parts[0]), float(parts[1]), float(parts[2])
                        # If rx differs from ry, or it has been randomly rotated by the ellipse generator
                        if abs(rx - ry) > 0.1 or rot != 0.0:
                            is_ellipse = True
                    except Exception:
                        pass
            if is_ellipse:
                predicted_shape = "ellipse"
            else:
                predicted_shape = "circle"
    else:
        if n_points == 3: predicted_shape = "triangle"
        elif n_points == 4:
            def dist(p1, p2):
                return math.hypot(p1[0]-p2[0], p1[1]-p2[1])
            L = [dist(pts[i], pts[(i+1)%4]) for i in range(4)]
            
            def eq(a, b): return abs(a - b) < 0.5
            
            is_kite = (eq(L[0], L[1]) and eq(L[2], L[3])) or (eq(L[0], L[3]) and eq(L[1], L[2]))
            
            def is_parallel(v1, v2):
                len1 = math.hypot(v1[0], v1[1])
                len2 = math.hypot(v2[0], v2[1])
                if len1 == 0 or len2 == 0: return True
                cross = v1[0]*v2[1] - v1[1]*v2[0]
                return abs(cross) / (len1 * len2) < 0.05
                
            v = [(pts[(i+1)%4][0]-pts[i][0], pts[(i+1)%4][1]-pts[i][1]) for i in range(4)]
            
            para02 = is_parallel(v[0], v[2])
            para13 = is_parallel(v[1], v[3])
            
            diag1 = dist(pts[0], pts[2])
            diag2 = dist(pts[1], pts[3])
            is_rect = para02 and para13 and eq(diag1, diag2)
            
            if is_rect:
                predicted_shape = "rectangle"
            elif is_kite:
                predicted_shape = "kite"
            elif para02 or para13:
                predicted_shape = "trapezoid"
            else:
                predicted_shape = "quadrilateral"
                
        elif n_points == 5: predicted_shape = "pentagon"
        elif n_points == 6: predicted_shape = "hexagon"
        elif n_points == 7: predicted_shape = "heptagon"
        elif n_points == 8: predicted_shape = "octagon"
        elif n_points == 9: predicted_shape = "nonagon"
        elif n_points == 10: predicted_shape = "decagon"
        elif n_points == 2: predicted_shape = "line"
        
    if not predicted_shape:
        return None
        
    option_dict = {opt[1].strip().lower(): opt[0] for opt in options_list}
    
    # Dataset specific quirk where true circles may be mapped to ellipse in the ground truth
    if predicted_shape == "ellipse":
        if "ellipse" in option_dict:
            return f"({option_dict['ellipse']})"
        elif "circle" in option_dict:
            return f"({option_dict['circle']})"
            
    if predicted_shape == "circle":
        if "circle" in option_dict:
            return f"({option_dict['circle']})"
        elif "ellipse" in option_dict:
            return f"({option_dict['ellipse']})"
            
    if predicted_shape in option_dict:
        return f"({option_dict[predicted_shape]})"
        
    # Safe hierarchical fallbacks for quadrilaterals ensuring correctness
    fallbacks = {
        "square": ["rectangle", "rhombus", "parallelogram", "trapezoid", "kite", "quadrilateral"],
        "rectangle": ["parallelogram", "trapezoid", "quadrilateral"],
        "rhombus": ["kite", "parallelogram", "trapezoid", "quadrilateral"],
        "parallelogram": ["trapezoid", "quadrilateral"],
        "trapezoid": ["quadrilateral"],
        "kite": ["quadrilateral"],
        "quadrilateral": []
    }
    
    if predicted_shape in fallbacks:
        for fb in fallbacks[predicted_shape]:
            if fb in option_dict:
                return f"({option_dict[fb]})"
                
    return None

  What was tried: An analysis of the failure cases reveals two major patterns:
1. **Miscounting Polygon Sides:** The LLM consistently miscounts the number of sides for $N \ge 5$ shapes (e.g., mislabeling a hexagon as a pentagon, or an octagon as a heptagon). 
2. **Ambiguity in Curves (Ellipses vs. Circles):** By default, the LLM maps all closed curves where the radii match ($r_x = r_y$) to "circle". However, in this dataset, these shapes often feature non-zero axis rotations (e.g. $285.4^\circ$), meaning they were generated as `ellipse` elements in SVG, so the true label is "ellipse".

Since the entire shape is geometrically defined in the SVG `d="..."` path, we can bypass the error-prone textual LLM reasoning and calculate the exact vertex counts, side lengths, and curve properties using standard geometric math in pure Python.

## HOW CHANGES WORK

- @interface-decorated functions with method=simulate use their DOCSTRING as
  the LLM prompt. Improving the docstring improves the LLM's output.
- Functions bound with method=direct run as pure Python.
- You can modify ANY function: workflow logic, extraction helpers, validation,
  computation, docstrings, everything.
- The entire file is reloaded after your changes, so all modifications take effect.

## STRATEGY — READ THIS CAREFULLY

Your changes are evaluated on ALL cases, not just the failures shown above.
A change that fixes 5 failures but breaks 15 working cases is a NET REGRESSION.
This has happened in previous iterations — learn from it.

**CONSERVATIVE APPROACH REQUIRED:**
- Changes to docstrings/prompts affect ALL cases that go through that code path.
  If you add instructions that help edge cases but confuse the LLM on common
  cases, accuracy drops. Be surgical.
- Changes to Python logic (extraction, validation, computation) are SAFER because
  they only fire when their conditions are met.
- Prefer adding NEW conditional branches or fallback logic over modifying
  existing working code paths.
- If a previous iteration was rolled back, DO NOT try a more aggressive version
  of the same idea. Try a completely different approach or a much smaller change.

**PRIORITIZE THESE CHANGE TYPES (safest to riskiest):**
1. Fix a bug in Python computation logic (zero risk to other cases)
2. Add a new fallback/recovery path for a specific failure pattern
3. Improve validation to catch and fix a category of extraction errors
4. Add a small, specific instruction to a docstring (low risk if well-scoped)
5. Rewrite extraction/prompt logic (HIGH RISK — only if you are very confident)

**AVOID:**
- Rewriting the extraction prompt wholesale — this breaks working cases
- Adding many instructions at once — impossible to know what helped vs hurt
- Changing unit conversion logic broadly — most values are already correct

## CRITICAL RULES

1. **DO NOT REWARD-HACK.** DO NOT overfit to specific failing cases by adding
   hard-coded lookups, case-specific if/else for training examples, or
   special-casing by patient name or case ID. Changes MUST generalize to
   unseen cases. A held-out test set will expose overfitting.

2. Make ONE targeted change per iteration. Don't rewrite everything at once.

3. Return the COMPLETE modified ptools_evolved.py inside a <ptools_file> block.
   Include ALL imports, ALL functions, everything. The block replaces the
   entire file.

4. Keep the file syntactically valid Python. Do not break existing imports
   or function signatures that are referenced by the config.
