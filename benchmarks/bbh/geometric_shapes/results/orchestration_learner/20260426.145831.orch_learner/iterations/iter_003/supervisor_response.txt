The task: given an SVG path element, identify which geometric shape it draws,
returning the correct multiple-choice letter (e.g. "(J)").
"""

# ── path normalization ──────────────────────────────────────────────────────

def _round_pt(x: float, y: float, decimals: int = 2) -> Tuple[float, float]:
    return (round(x, decimals), round(y, decimals))

def normalize_path(commands: List[str]) -> List[str]:
    """Rearrange SVG commands into continuous chains where possible.

    Preserves all command types (L, A, C, etc.). Connects subpaths that
    share endpoints to remove redundant M commands.
    """
    if not commands:
        return commands

    parsed = []
    current_pt = None
    for cmd in commands:
        cmd_str = cmd.strip()
        if not cmd_str:
            continue
            
        cmd_type = cmd_str[0].upper()
        nums = [float(m) for m in re.findall(r'-?\d*\.?\d+', cmd_str)]
        
        try:
            if cmd_type == 'Z':
                end_pt = current_pt
            elif len(nums) >= 2:
                end_pt = _round_pt(nums[-2], nums[-1])
            else:
                end_pt = current_pt
        except Exception:
            end_pt = current_pt
            
        start_pt = end_pt if cmd_type == 'M' else current_pt
        
        parsed.append({
            'text': cmd_str,
            'type': cmd_type,
            'start': start_pt,
            'end': end_pt
        })
        if end_pt is not None:
            current_pt = end_pt

    subpaths = []
    curr_sp = []
    for p in parsed:
        if p['type'] == 'M':
            if curr_sp:
                subpaths.append(curr_sp)
            curr_sp = [p]
        else:
            if not curr_sp:
                curr_sp = [{'text': f"M {p['start'][0]},{p['start'][1]}", 'type': 'M', 'start': p['start'], 'end': p['start']}]
            curr_sp.append(p)
    if curr_sp:
        subpaths.append(curr_sp)

    if not subpaths:
        return commands

    chained = []
    used = [False] * len(subpaths)
    
    for i in range(len(subpaths)):
        if used[i]:
            continue
        used[i] = True
        current_chain = list(subpaths[i])
        
        while True:
            tip_pt = current_chain[-1]['end']
            if tip_pt is None:
                break
                
            extended = False
            for j in range(len(subpaths)):
                if not used[j]:
                    start_pt = subpaths[j][0]['end']
                    if start_pt is not None and tip_pt == start_pt:
                        used[j] = True
                        current_chain.extend(subpaths[j][1:])
                        extended = True
                        break
            if not extended:
                break
        chained.append(current_chain)
        
    result = []
    for chain in chained:
        for cmd in chain:
            result.append(cmd['text'])
            
    return result

# ── fast pure python equivalents ─────────────────────────────────────────────

def _decompose_path_py(path: str) -> List[str]:
    tokens = re.split(r'(?=[MmLlHhVvCcSsQqTtAaZz])', path)
    return [t.strip() for t in tokens if t.strip()]

def _describe_command_py(command: str, previous_command: Optional[str] = None, side_idx: int = 0) -> str:
    cmd = command.strip()
    if not cmd:
        return ""
    cmd_type = cmd[0].upper()
    nums = [float(m) for m in re.findall(r'-?\d*\.?\d+', cmd)]
    
    prefix = f"Side {side_idx}: " if side_idx > 0 else ""
    
    if cmd_type == 'M' and len(nums) >= 2:
        return f"Move to ({nums[0]}, {nums[1]})"
    elif cmd_type == 'L' and len(nums) >= 2:
        end_x, end_y = nums[-2], nums[-1]
        if previous_command:
            prev_nums = [float(m) for m in re.findall(r'-?\d*\.?\d+', previous_command)]
            if len(prev_nums) >= 2:
                start_x, start_y = prev_nums[-2], prev_nums[-1]
                length = math.hypot(end_x - start_x, end_y - start_y)
                return f"{prefix}Draw a line from ({start_x}, {start_y}) to ({end_x}, {end_y}) of length {length:.2f}"
        return f"{prefix}Draw a line to ({end_x}, {end_y})"
    elif cmd_type == 'A' and len(nums) >= 7:
        return f"{prefix}Draw an elliptical arc to ({nums[-2]}, {nums[-1]}) with radii {nums[0]}, {nums[1]}"
    else:
        return f"Draw path {cmd}"

def _compute_angle_py(prev_cmd: str, curr_cmd: str, next_cmd: str) -> str:
    if not (curr_cmd.strip().upper().startswith('L') and next_cmd.strip().upper().startswith('L')):
        return ""
    try:
        def get_pt(cmd):
            nums = [float(m) for m in re.findall(r'-?\d*\.?\d+', cmd)]
            return nums[-2], nums[-1]
        
        p1 = get_pt(prev_cmd)
        p2 = get_pt(curr_cmd)
        p3 = get_pt(next_cmd)
        
        v1 = (p1[0] - p2[0], p1[1] - p2[1])
        v2 = (p3[0] - p2[0], p3[1] - p2[1])
        
        dot = v1[0]*v2[0] + v1[1]*v2[1]
        mag1 = math.hypot(*v1)
        mag2 = math.hypot(*v2)
        
        if mag1 == 0 or mag2 == 0:
            return ""
            
        val = dot / (mag1 * mag2)
        val = max(-1.0, min(1.0, val))
        angle = math.degrees(math.acos(val))
        return f"Angle between segments: {angle:.1f} degrees"
    except Exception:
        return ""

@interface
def describe_shape(annotated_commands: List[str]) -> str:
    """Given the full list of commands with lengths and angles interspersed,
    describe what geometric shape the path forms.
    
    CRITICAL RULES:
    1. Count the number of sides (labeled as Side 1, Side 2, etc.). 
       1 side = line
       3 sides = triangle
       4 sides = quadrilateral (can be kite, rectangle, square, trapezoid)
       5 sides = pentagon
       6 sides = hexagon
       7 sides = heptagon
       8 sides = octagon
    2. For 4-sided shapes (quadrilaterals):
       - If it has 2 pairs of equal adjacent sides (e.g., lengths A, B, B, A), it is a 'kite'.
       - If it has exactly 4 equal sides and 90 deg angles, it is a 'rectangle' or 'square'.
       - If it has exactly 90.0 deg angles but sides are not all equal, it is a 'rectangle'.
       - If opposite sides are approximately equal but angles are NOT exactly 90.0 (e.g., 89.9, 90.1), it is a parallelogram. If 'parallelogram' is not an option, select 'trapezoid'.
       - If it has exactly 1 pair of parallel sides (angles sum to 180 on one side), it is a 'trapezoid'.
    3. For curves:
       - If the path has two elliptical arc (A) commands forming a closed loop, you MUST identify it as an 'ellipse' (even if radii are equal, prefer 'ellipse' over 'circle').
       - If the path has exactly 2 straight lines of equal length and 1 arc, it is a 'sector'.
    """
    ...

# ── hand-coded workflow ───────────────────────────────────────────────────────

def geometric_shapes_workflow(input: str) -> str:
    """Hand-coded workflow implementing identify_shape.

    To use:
        ptools.identify_shape.method=direct
        ptools.identify_shape.fn=ptools.geometric_shapes_workflow
    """
    path, options = extract_path_and_options(input)
    commands = normalize_path(_decompose_path_py(path))

    if not commands:
        return "(A)"

    side_counter = 1
    descriptions = []
    for i in range(len(commands)):
        is_side = not commands[i].strip().upper().startswith('M')
        desc = _describe_command_py(
            commands[i], 
            previous_command=commands[i - 1] if i > 0 else None,
            side_idx=side_counter if is_side else 0
        )
        if is_side:
            side_counter += 1
        descriptions.append(desc)

    annotated = [descriptions[0]]
    for i in range(1, len(descriptions)):
        annotated.append(descriptions[i])
        
        # Interleave angle directly AFTER the incoming segment has been drawn
        if i + 1 < len(commands):
            angle = _compute_angle_py(commands[i - 1], commands[i], commands[i + 1])
            if angle:
                annotated.append(angle)
        
        # Connect last and first segment angle if closed polygon
        elif len(commands) > 2 and commands[-1].strip().upper().startswith('L') and commands[0].strip().upper().startswith('M'):
            def get_pt(cmd):
                nums = [float(m) for m in re.findall(r'-?\d*\.?\d+', cmd)]
                return nums[-2], nums[-1]
            try:
                start_pt = get_pt(commands[0])
                end_pt = get_pt(commands[-1])
                if math.hypot(start_pt[0]-end_pt[0], start_pt[1]-end_pt[1]) < 1e-2:
                    angle = _compute_angle_py(commands[-2], commands[-1], commands[1])
                    if angle:
                        annotated.append(angle)
            except Exception:
                pass

    description = describe_shape(annotated)
    return select_option(description, options)
