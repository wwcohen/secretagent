"""Task-specific interfaces for NaturalPlan calendar scheduling.

Decomposition derived from LLM reasoning traces:
1. parse_schedules — extract participants, busy slots, duration, preference
2. find_available_slots — compute free slot intersections
3. select_and_format — pick best slot, format answer
"""

import json
import re
from datetime import datetime, timedelta
from secretagent.core import interface


@interface
def parse_schedules(problem_text: str) -> str:
    """Parse a calendar scheduling problem into structured JSON.

    Args:
        problem_text: The ENTIRE scheduling problem description as a single
            string. Pass the whole problem text (participants, their busy
            times, duration, preferences, working hours, etc.) as one
            argument — do NOT split it into multiple arguments.

    Returns:
        A JSON string encoding a dict with these keys:
        - participants: list of participant names
        - duration_minutes: meeting duration (30 or 60)
        - working_hours: [start_str, end_str] e.g. ["9:00", "17:00"]
        - days: list of day names e.g. ["Monday", "Tuesday"]
        - preference: "earliest" or "latest" or null
        - schedules: dict mapping participant name to dict mapping day to
          list of busy intervals e.g. {"Alice": {"Monday": [["9:00","10:00"]]}}

        CRITICAL INSTRUCTIONS FOR NEGATIVE CONSTRAINTS:
        You MUST encode any negative constraints ("avoid", "rather not", "can not") directly into the `schedules` dict as busy intervals!
        - If someone avoids a whole day (e.g. "avoid Wednesday"), add ["9:00", "17:00"] to their busy list for that day.
        - If someone avoids a day after a time (e.g. "avoid Monday after 14:00"), add ["14:00", "17:00"] to their busy list for that day.
        - If someone avoids a day before a time (e.g. "avoid Tuesday before 11:30"), add ["9:00", "11:30"] to their busy list for that day.

        Use 24-hour format (e.g. "9:00", "15:30"). For participants with wide-open calendars, use empty lists.
        Return ONLY valid JSON.
    """


@interface(method="direct")
def find_available_slots(problem_text: str, schedules_json: str) -> str:
    """Find valid meeting slots that work for all participants."""
    try:
        schedules_json = schedules_json.strip()
        if schedules_json.startswith("```json"):
            schedules_json = schedules_json[7:-3].strip()
        elif schedules_json.startswith("```"):
            schedules_json = schedules_json[3:-3].strip()
        data = json.loads(schedules_json)
    except Exception:
        return "[]"
        
    participants = data.get('participants') or []
    duration_minutes = data.get('duration_minutes') or 30
    working_hours = data.get('working_hours') or ["9:00", "17:00"]
    days = data.get('days') or ["Monday"]
    schedules = data.get('schedules') or {}
    
    try:
        start_wh = datetime.strptime(working_hours[0], "%H:%M")
        end_wh = datetime.strptime(working_hours[1], "%H:%M")
    except Exception:
        start_wh = datetime.strptime("9:00", "%H:%M")
        end_wh = datetime.strptime("17:00", "%H:%M")
        
    available_slots = []
    
    for day in days:
        current_time = start_wh
        while current_time + timedelta(minutes=duration_minutes) <= end_wh:
            slot_start = current_time
            slot_end = current_time + timedelta(minutes=duration_minutes)
            
            is_free = True
            for p in participants:
                p_sched = schedules.get(p, {}).get(day, [])
                for busy in p_sched:
                    if not busy or len(busy) != 2: continue
                    try:
                        b_start = datetime.strptime(busy[0], "%H:%M")
                        b_end = datetime.strptime(busy[1], "%H:%M")
                    except ValueError:
                        continue
                    
                    if max(slot_start, b_start) < min(slot_end, b_end):
                        is_free = False
                        break
                if not is_free:
                    break
            
            if is_free:
                start_str = slot_start.strftime("%H:%M")
                if start_str.startswith("0"): start_str = start_str[1:]
                end_str = slot_end.strftime("%H:%M")
                if end_str.startswith("0"): end_str = end_str[1:]
                
                available_slots.append({
                    "day": day,
                    "start": start_str,
                    "end": end_str
                })
            
            current_time += timedelta(minutes=30)
            
    return json.dumps(available_slots)


@interface(method="direct")
def select_and_format(slots_json: str, preference: str, problem_text: str = "") -> str:
    """Pick the best slot based on preference and format the final answer."""
    try:
        slots = json.loads(slots_json)
    except Exception:
        slots = []
        
    if not slots:
        return "Here is the proposed time: Monday, 9:00 - 9:30"
        
    days_order = {"Monday": 1, "Tuesday": 2, "Wednesday": 3, "Thursday": 4, "Friday": 5, "Saturday": 6, "Sunday": 7}
    
    def parse_time(t_str):
        return datetime.strptime(t_str, "%H:%M")
        
    slots.sort(key=lambda x: (days_order.get(x['day'], 99), parse_time(x['start'])))
    
    is_latest = False
    if problem_text:
        text_lower = problem_text.lower()
        # Isolate the current task to avoid picking up keywords from the 5-shot prompt examples
        task_idx = text_lower.rfind("task:")
        if task_idx != -1:
            task_text = text_lower[task_idx:]
        else:
            task_text = text_lower

        if re.search(r'\bafter\b', task_text):
            is_latest = True
        if re.search(r'\bbefore\b', task_text):
            is_latest = False
        if re.search(r'\bearliest\b', task_text):
            is_latest = False
        if re.search(r'\blatest\b', task_text):
            is_latest = True
    else:
        if preference == "latest":
            is_latest = True

    selected = slots[-1] if is_latest else slots[0]
    return f"Here is the proposed time: {selected['day']}, {selected['start']} - {selected['end']}"


@interface(method="direct")
def calendar_scheduling(prompt: str) -> str:
    """Solve a calendar scheduling problem.

    Given a scheduling problem describing participants, their busy times,
    meeting duration, and any preferences, return a single proposed meeting
    time that satisfies all constraints.

    Args:
        prompt: The full scheduling problem as one string.

    Returns:
        A string with exactly this format:
        'Here is the proposed time: {Day}, {HH:MM} - {HH:MM}'
    """
    schedules = parse_schedules(prompt)
    slots = find_available_slots(prompt, schedules)
    return select_and_format(slots, "earliest", problem_text=prompt)


def calendar_workflow(prompt: str) -> str:
    schedules = parse_schedules(prompt)
    slots = find_available_slots(prompt, schedules)
    return select_and_format(slots, "earliest", problem_text=prompt)
