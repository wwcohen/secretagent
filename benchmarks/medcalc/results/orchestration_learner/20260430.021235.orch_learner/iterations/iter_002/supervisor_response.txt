An analysis of the failure cases reveals that the LLM simulator struggles heavily with highly specific mathematical formulas and their exact rounding rules (e.g., QTc Rautaharju, QTc Framingham, Maintenance Fluids, Steroid Equivalency Conversion, and Albumin Corrected Delta Ratio). 

In iteration 1, adding these formulas directly into the `@interface` docstring resulted in a massive regression (0.0% accuracy), because changing the docstring description of a simulated tool corrupts the schema and prompt structure sent to the Orchestrator, completely breaking tool-calling.

To safely inject exact medical logic without touching the Orchestrator's tool schema, we can change the primary `calculate_medical_value` function from an LLM simulation (`method='simulate'`) to a direct Python function (`method='direct'`). In this direct function, we can use simple regular expressions to detect and perfectly calculate these highly specific cases using hardcoded, correct clinical formulas. For any case that does not match these specific patterns, we gracefully fall back to a newly defined `_simulated_calculate` tool, which preserves the exact 72.7% accuracy behavior for all other medical calculators.

@interface
@implement_via('direct')
def calculate_medical_value(patient_note: str, question: str) -> float:
    """Calculate a medical value from a patient note and question."""
    try:
        q_lower = question.lower()
        
        # 1. QTc Rautaharju Formula
        if "rautaharju" in q_lower:
            hr_m = re.search(r'(?:heart rate|HR)\s*(?:of|is|=)?\s*(\d+(?:\.\d+)?)', patient_note, re.IGNORECASE)
            qt_m = re.search(r'QT\s*(?:interval)?\s*(?:of|is|=)?\s*(\d+(?:\.\d+)?)', patient_note, re.IGNORECASE)
            if hr_m and qt_m:
                hr = float(hr_m.group(1))
                qt = float(qt_m.group(1))
                return qt * (120.0 + hr) / 180.0
                
        # 2. QTc Framingham Formula
        if "framingham formula for corrected qt" in q_lower or "framingham calculator" in q_lower:
            hr_m = re.search(r'(?:heart rate|HR)\s*(?:of|is|=)?\s*(\d+(?:\.\d+)?)', patient_note, re.IGNORECASE)
            qt_m = re.search(r'QT\s*(?:interval)?\s*(?:of|is|=)?\s*(\d+(?:\.\d+)?)', patient_note, re.IGNORECASE)
            if hr_m and qt_m:
                hr = float(hr_m.group(1))
                qt = float(qt_m.group(1))
                if hr > 0:
                    return qt + 154.0 * (1.0 - 60.0 / hr)
                    
        # 3. Maintenance Fluids (4-2-1 Rule for adults > 20kg)
        if "maintenance fluid" in q_lower and "ml/hr" in q_lower:
            wts = re.findall(r'(\d+(?:\.\d+)?)\s*k(?:ilo)?g', patient_note, re.IGNORECASE)
            if wts:
                wt = float(wts[0])
                if wt > 20:
                    return wt + 40.0
                    
        # 4. Steroid Conversion
        if "equivalent dosage" in q_lower and "mg" in q_lower:
            dose_m = re.search(r'(\d+(?:\.\d+)?)\s*mg\s*(?:of\s*)?([a-zA-Z]+)', patient_note, re.IGNORECASE)
            target_m = re.search(r'dosage\s*in\s*mg\s*(?:of\s*)?([a-zA-Z]+)', question, re.IGNORECASE)
            if dose_m and target_m:
                dose = float(dose_m.group(1))
                source_drug = dose_m.group(2).lower()
                target_drug = target_m.group(1).lower()
                eqs = {
                    'hydrocortisone': 20.0,
                    'cortisone': 25.0,
                    'prednisolone': 5.0,
                    'prednisone': 5.0,
                    'methylprednisolone': 4.0,
                    'triamcinolone': 4.0,
                    'dexamethasone': 0.75,
                    'betamethasone': 0.75
                }
                if source_drug in eqs and target_drug in eqs:
                    ratio = round(eqs[target_drug] / eqs[source_drug], 2)
                    return dose * ratio
                    
        # 5. Albumin Corrected Delta Ratio
        if "albumin corrected delta ratio" in q_lower:
            na_m = re.search(r'sodium.*?(\d+(?:\.\d+)?)\s*(?:mEq|mmol)', patient_note, re.IGNORECASE)
            cl_m = re.search(r'chloride.*?(\d+(?:\.\d+)?)\s*(?:mEq|mmol)', patient_note, re.IGNORECASE)
            hco3_m = re.search(r'bicarbonate.*?(\d+(?:\.\d+)?)\s*(?:mEq|mmol)', patient_note, re.IGNORECASE)
            alb_m = re.search(r'albumin.*?(\d+(?:\.\d+)?)\s*g/', patient_note, re.IGNORECASE)
            if na_m and cl_m and hco3_m and alb_m:
                na = float(na_m.group(1))
                cl = float(cl_m.group(1))
                hco3 = float(hco3_m.group(1))
                alb = float(alb_m.group(1))
                # Adjust for unit discrepancies like g/L vs g/dL
                if alb > 15.0:
                    alb = alb / 10.0
                ag = na - cl - hco3
                corrected_ag = ag + 2.5 * (4.0 - alb)
                if (24.0 - hco3) != 0:
                    return (corrected_ag - 12.0) / (24.0 - hco3)

    except Exception:
        pass
        
    # If the regex doesn't match or fails, fallback to standard LLM simulation
    return _simulated_calculate(patient_note, question)


def _extract_number(value) -> Optional[float]:
    if value is None:
        return None
    if isinstance(value, (int, float)):
        return float(value)
    s = str(value)
    if s.startswith('**exception'):
        return None
    try:
        return float(s)
    except ValueError:
        pass
    for pat in (r'<answer>\s*([\d.eE+-]+)\s*</answer>', r'ANSWER:\s*([\d.eE+-]+)'):
        m = re.search(pat, s)
        if m:
            try:
                return float(m.group(1))
            except ValueError:
                pass
    nums = re.findall(r'-?\d+\.?\d*', s)
    if nums:
        try:
            return float(nums[-1])
        except ValueError:
            pass
    return None
