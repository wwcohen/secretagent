def _extract_number_local(value) -> Optional[float]:
    """Local copy of expt._extract_number to avoid circular import."""
    if value is None:
        return None
    if isinstance(value, (int, float)):
        return float(value)
    s = str(value)
    if s.startswith('**exception'):
        return None
    try:
        return float(s)
    except ValueError:
        pass
    m = re.search(r'<answer>\s*([\d.eE+-]+)\s*</answer>', s)
    if m:
        try:
            return float(m.group(1))
        except ValueError:
            pass
    m = re.search(r'ANSWER:\s*([\d.eE+-]+)', s)
    if m:
        try:
            return float(m.group(1))
        except ValueError:
            pass
    nums = re.findall(r'-?\d+\.?\d*', s)
    if nums:
        try:
            return float(nums[-1])
        except ValueError:
            pass
    return None

def _build_descriptive_fields(calc_name: str) -> list[str]:
    """Build descriptive field names from calculator docstring."""
    import calculator_simple
    doc = calculator_simple.get_calculator_docstring(calc_name) or ''
    sigs = calculator_simple.get_calculator_signatures()
    sig = sigs.get(calc_name, {})
    all_fields = sig.get('required', []) + sig.get('optional', [])

    field_desc = {}
    for line in doc.split('\n'):
        line = line.strip()
        match = re.match(r'(\w+):\s+(.+)', line)
        if match and match.group(1) in all_fields:
            field_desc[match.group(1)] = match.group(2).strip()

    result = []
    for f in all_fields:
        if f in field_desc:
            result.append(f'{f}: {field_desc[f]}')
        else:
            result.append(f)
    return result


def workflow(patient_note: str, question: str) -> Any:
    """L4 workflow: Python-orchestrated with ptools interfaces + LLM extraction."""
    import calculator_simple
    from calculators import identify_calculator as python_identify

    signatures = calculator_simple.get_calculator_signatures()
    available = list(signatures.keys())

    # ---- Stage 1: identify_calculator (ptools interface) ----
    calc_name = None
    try:
        result = identify_calculator(question, available)
        
        # Soften markdown codeblock boundaries for resilient JSON loading
        if isinstance(result, str):
            m = re.search(r'\{.*\}', result, re.DOTALL)
            if m:
                try: result = json.loads(m.group())
                except Exception: pass
                
        if isinstance(result, dict):
            calc_name = result.get("calculator_name")
            if calc_name and calc_name not in signatures:
                calc_lower = calc_name.lower()
                for sig_name in signatures:
                    if calc_lower in sig_name.lower() or sig_name.lower() in calc_lower:
                        calc_name = sig_name
                        break
                else:
                    calc_name = None
    except Exception:
        pass

    if not calc_name:
        pattern = python_identify(question)
        if pattern:
            for name, spec in calculator_simple.CALCULATOR_REGISTRY.items():
                if isinstance(spec, calculator_simple.CalculatorSpec):
                    if pattern.lower() in spec.name.lower():
                        calc_name = spec.name
                        break

    # ---- Stage 1.5: Pure Python Date Intercept (zero LLM coercion crashes) ----
    if (calc_name and calc_name in ["Estimated Due Date", "Estimated Date of Conception", "Estimated Gestational Age"]) or _is_date_question(question):
        import re
        from dateutil import parser
        from datetime import timedelta
        text = patient_note + " " + question
        dates = re.findall(r'\d{1,2}/\d{1,2}/\d{2,4}', text)
        if dates:
            try:
                lmp_date = parser.parse(dates[0])
                curr_date = parser.parse(dates[1]) if len(dates) > 1 else None
                cycle_match = re.search(r'cycle.*?(\d+)\s*days', text, re.IGNORECASE)
                cycle = int(cycle_match.group(1)) if cycle_match else 28
                
                q_lower = question.lower()
                c_name = (calc_name or "").lower()
                if "conception" in c_name or "conception" in q_lower:
                    edc = lmp_date + timedelta(days=14 + (cycle - 28))
                    return edc.strftime("%m/%d/%Y")
                elif "due date" in c_name or "due date" in q_lower:
                    edd = lmp_date + timedelta(days=280 + (cycle - 28))
                    return edd.strftime("%m/%d/%Y")
                elif "gestational age" in c_name or "gestational age" in q_lower:
                    if curr_date:
                        diff = curr_date - lmp_date
                        weeks = diff.days // 7
                        days = diff.days % 7
                        return f"('{weeks} weeks', '{days} days')"
            except Exception:
                pass

    # If identification fails, return None to trigger safe orchestrator fallback
    if not calc_name:
        return None

    # ---- Stage 2: Extract values (LLM with calculator-specific context) ----
    sig = signatures.get(calc_name, {})
    required = sig.get("required", [])
    optional = sig.get("optional", [])

    extracted = _extract_values_two_stage(
        patient_note, calc_name, required, optional
    )

    # ---- Stage 3: Validate + compute_calculation (ptools interface) ----
    is_valid, missing, cleaned = _validate_extracted_values(extracted, calc_name)

    if missing and not is_valid:
        repaired = _repair_extraction(patient_note, calc_name, cleaned, missing)
        is_valid, missing, cleaned = _validate_extracted_values(repaired, calc_name)

    q_lower = question.lower()

    # Self-healing math overrides that bypass strict validation constraints
    if calc_name == "Maintenance Fluids (4-2-1 Rule)":
        wt = cleaned.get("weight_kg")
        if wt is None:
            import re
            m = re.search(r'(\d+(?:\.\d+)?)\s*kg', patient_note.lower())
            if m: wt = float(m.group(1))
        if wt is not None:
            if wt <= 10: return float(4 * wt)
            elif wt <= 20: return float(40 + 2 * (wt - 10))
            else: return float(60 + 1 * (wt - 20))
            
    if calc_name == "Morphine Milligram Equivalents (MME)":
        import re
        mme_total = 0.0
        factors = {
            'morphine': 1.0, 'oxymorphone': 3.0, 'tapentadol': 0.4,
            'oxycodone': 1.5, 'hydrocodone': 1.0, 'hydromorphone': 4.0,
            'methadone': 3.0, 'codeine': 0.15, 'tramadol': 0.1
        }
        text = (patient_note + " " + question).lower()
        for drug, factor in factors.items():
            pattern = rf'(\d+(?:\.\d+)?)\s*mg\s*of\s*{drug}.*?(\d+)\s*times\s*a\s*day'
            for dose, times in re.findall(pattern, text):
                mme_total += float(dose) * float(times) * factor
        if mme_total > 0:
            return float(mme_total)

    if is_valid:
        if calc_name == "Albumin Corrected Anion Gap" or "delta gap" in q_lower or "delta ratio" in q_lower:
            if "sodium" in cleaned and "chloride" in cleaned and "bicarbonate" in cleaned:
                ag = cleaned["sodium"] - cleaned["chloride"] - cleaned["bicarbonate"]
                alb = cleaned.get("albumin")
                if alb is None:
                    import re
                    m = re.search(r'albumin[^\d]*(\d+(?:\.\d+)?)', patient_note.lower())
                    if m: alb = float(m.group(1))
                    else: alb = 4.0
                ag_corr = ag + 2.5 * (4.0 - alb)
                if "delta gap" in q_lower:
                    return float(ag_corr - 12.0)
                if "delta ratio" in q_lower:
                    return float((ag_corr - 12.0) / (24.0 - cleaned["bicarbonate"]))
                return float(ag_corr)

        if calc_name == "QTc (Rautaharju)":
            if "qt_msec" in cleaned and "heart_rate" in cleaned:
                return float(cleaned["qt_msec"] * (120 + cleaned["heart_rate"]) / 180.0)

        if calc_name in ["Centor Score (McIsaac)", "Centor Score (Modified/McIsaac)", "Centor Score"]:
            if "centor score" in q_lower and "mcisaac" not in q_lower:
                score = sum([
                    1 if cleaned.get('fever') else 0,
                    1 if cleaned.get('absence_of_cough') else 0,
                    1 if cleaned.get('tonsillar_exudate') else 0,
                    1 if cleaned.get('tender_lymphadenopathy') else 0
                ])
                return float(score)

        if calc_name == "Steroid Conversion" and "target_steroid" not in cleaned:
            import re
            m = re.search(r'equivalent dosage.*?of\s*([a-zA-Z]+)', question, re.IGNORECASE)
            if m:
                cleaned["target_steroid"] = m.group(1).lower()

        try:
            result = compute_calculation(calc_name, cleaned)
            if isinstance(result, dict) and result.get('result') is not None:
                return result['result']
        except Exception:
            pass

    return None

def _validate_extracted_values(
    extracted: dict, calculator_name: str
) -> tuple[bool, list[str], dict]:
    """Validate and clean extracted values."""
    import calculator_simple
    import re

    flattened = {}
    for key, value in extracted.items():
        if isinstance(value, dict):
            for subkey, subvalue in value.items():
                flattened[subkey] = subvalue
        else:
            flattened[key] = value

    cleaned = {}
    for key, value in flattened.items():
        if isinstance(value, str):
            val_lower = value.lower().strip()
            if val_lower in ("true", "yes", "1", "present", "positive"):
                cleaned[key] = True
            elif val_lower in ("false", "no", "0", "absent", "negative"):
                cleaned[key] = False
            elif val_lower in ("m", "male", "man", "he"):
                cleaned[key] = "male"
            elif val_lower in ("f", "female", "woman", "she"):
                cleaned[key] = "female"
            else:
                try:
                    cleaned[key] = float(value)
                except ValueError:
                    m = re.match(r'^(-?\d+(?:\.\d+)?)\s*[a-zA-Z/%]+', val_lower)
                    if m and "/" not in val_lower and "-" not in val_lower[1:]:
                        cleaned[key] = float(m.group(1))
                    else:
                        # Extract basic digits safely, dodging dates (/) and times (:)
                        m2 = re.search(r'-?\d+(?:\.\d+)?', val_lower)
                        if m2 and "/" not in val_lower and ":" not in val_lower:
                            cleaned[key] = float(m2.group())
                        else:
                            cleaned[key] = value.strip()
        elif value is not None:
            cleaned[key] = value

    # Fix common magnitude scaling mistake for albumin
    if 'albumin' in cleaned and isinstance(cleaned['albumin'], (int, float)):
        if cleaned['albumin'] > 15:
            cleaned['albumin'] /= 10.0

    signatures = calculator_simple.get_calculator_signatures()
    sig = signatures.get(calculator_name, {})
    required = sig.get("required", [])
    
    missing = [v for v in required if v not in cleaned or cleaned[v] is None]
    return len(missing) == 0, missing, cleaned

## HOW CHANGES WORK

- @interface-decorated functions with method=simulate use their DOCSTRING as
  the LLM prompt. Improving the docstring improves the LLM's output.
- Functions bound with method=direct run as pure Python.
- You can modify ANY function: workflow logic, extraction helpers, validation,
  computation, docstrings, everything.
- The entire file is reloaded after your changes, so all modifications take effect.

## STRATEGY — READ THIS CAREFULLY

Your changes are evaluated on ALL cases, not just the failures shown above.
A change that fixes 5 failures but breaks 15 working cases is a NET REGRESSION.
This has happened in previous iterations — learn from it.

**CONSERVATIVE APPROACH REQUIRED:**
- Changes to docstrings/prompts affect ALL cases that go through that code path.
  If you add instructions that help edge cases but confuse the LLM on common
  cases, accuracy drops. Be surgical.
- Changes to Python logic (extraction, validation, computation) are SAFER because
  they only fire when their conditions are met.
- Prefer adding NEW conditional branches or fallback logic over modifying
  existing working code paths.
- If a previous iteration was rolled back, DO NOT try a more aggressive version
  of the same idea. Try a completely different approach or a much smaller change.

**PRIORITIZE THESE CHANGE TYPES (safest to riskiest):**
1. Fix a bug in Python computation logic (zero risk to other cases)
2. Add a new fallback/recovery path for a specific failure pattern
3. Improve validation to catch and fix a category of extraction errors
4. Add a small, specific instruction to a docstring (low risk if well-scoped)
5. Rewrite extraction/prompt logic (HIGH RISK — only if you are very confident)

**AVOID:**
- Rewriting the extraction prompt wholesale — this breaks working cases
- Adding many instructions at once — impossible to know what helped vs hurt
- Changing unit conversion logic broadly — most values are already correct

## CRITICAL RULES

1. **DO NOT REWARD-HACK.** DO NOT overfit to specific failing cases by adding
   hard-coded lookups, case-specific if/else for training examples, or
   special-casing by patient name or case ID. Changes MUST generalize to
   unseen cases. A held-out test set will expose overfitting.
