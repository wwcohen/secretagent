def _extract_number(value) -> Optional[float]:
    if value is None:
        return None
    if isinstance(value, (int, float)):
        return float(value)
    s = str(value)
    if s.startswith('**exception'):
        return None
    try:
        return float(s)
    except ValueError:
        pass
        
    for pat in (r'<answer>\s*([\d.eE+-]+)\s*</answer>', r'ANSWER:\s*([\d.eE+-]+)'):
        m = re.search(pat, s, re.IGNORECASE)
        if m:
            try:
                return float(m.group(1))
            except ValueError:
                pass
                
    for pat in (r'total.*?score.*?(?:is|:|=)\s*([\d.eE+-]+)', r'score.*?(?:is|:|=)\s*([\d.eE+-]+)'):
        matches = re.findall(pat, s, re.IGNORECASE)
        if matches:
            try:
                return float(matches[-1])
            except ValueError:
                pass
                
    nums = re.findall(r'-?\d+\.?\d*', s)
    if nums:
        try:
            return float(nums[-1])
        except ValueError:
            pass
    return None
