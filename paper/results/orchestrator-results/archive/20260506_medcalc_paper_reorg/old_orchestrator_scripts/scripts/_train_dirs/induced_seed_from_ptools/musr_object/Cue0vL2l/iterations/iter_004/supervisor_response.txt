def extract_index(belief_state: str, choices: list) -> int:
    """Robust extraction to handle soft matches and avoid -1 returns."""
    loc = ""
    try:
        match = re.search(r'\{[^{}]*\}', str(belief_state), re.DOTALL)
        if match:
            data = json.loads(match.group(0))
            if "believed_location" in data:
                loc = str(data["believed_location"]).strip().lower()
    except Exception:
        pass

    c_clean = [str(c).strip().lower() for c in choices]

    if loc:
        # Exact match
        for i, c in enumerate(c_clean):
            if c == loc: return i
            
        # Exact match excluding punctuation
        loc_no_punct = re.sub(r'[^\w\s]', '', loc).strip()
        for i, c in enumerate(c_clean):
            c_no_punct = re.sub(r'[^\w\s]', '', c).strip()
            if c_no_punct and c_no_punct == loc_no_punct: return i

        # Substring matching
        for i, c in enumerate(c_clean):
            if c in loc: return i
        for i, c in enumerate(c_clean):
            if loc in c: return i

        # Word overlap matching
        loc_words = set(re.findall(r'\b\w+\b', loc))
        best_i = -1
        max_overlap = 0
        for i, c in enumerate(c_clean):
            c_words = set(re.findall(r'\b\w+\b', c))
            overlap = len(loc_words & c_words)
            if overlap > max_overlap:
                max_overlap = overlap
                best_i = i
        if best_i != -1:
            return best_i

    # Fallback to searching the entire raw text
    belief_lower = str(belief_state).lower()
    best_i = -1
    best_pos = -1
    for i, c in enumerate(c_clean):
        pos = belief_lower.rfind(c)
        if pos > best_pos:
            best_pos = pos
            best_i = i
            
    return best_i if best_i != -1 else 0
